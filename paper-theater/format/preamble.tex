% Theorems and discs
% =====================================================================
% amsthm is loaded for \newtheoremstyle (the disc design). newtxmath
% already defines \openbox, so neutralise it first.
\let\openbox\relax
\usepackage{amsthm}

% --- status colours ---
\definecolor{stverified}{RGB}{34,139,34}    % green      -> verified    (v)
\definecolor{stplausible}{RGB}{31,119,180}  % blue       -> plausible   (p)
\definecolor{stheuristic}{RGB}{230,140,0}   % orange     -> heuristic   (h)
\definecolor{stconjectured}{RGB}{200,30,30} % red        -> conjectured (c)
\definecolor{ststated}{RGB}{105,105,105}    % gray       -> stated      (s)  script mode
\definecolor{stoutside}{RGB}{170,170,170}   % light gray -> outside     (o)  script mode

% Internal helper: a colour-filled bullet on the baseline. No TikZ.
\newcommand{\discbullet}[1]{{\textcolor{#1}{\raisebox{0.05ex}{\large$\bullet$}}}}

% THE one public macro: \disc{X}, X in {v,p,h,c,s,o}. One alphabet
% everywhere — headers, skeleton rows, and the theorem note slot all call
% \disc{...}. Empty/unknown key -> nothing (graceful).
\newcommand{\disc}[1]{%
  \ifx\\#1\\%
  \else
    \begingroup
    \def\tmp{#1}%
    \def\kv{v}\ifx\tmp\kv\discbullet{stverified}\fi
    \def\kp{p}\ifx\tmp\kp\discbullet{stplausible}\fi
    \def\kh{h}\ifx\tmp\kh\discbullet{stheuristic}\fi
    \def\kc{c}\ifx\tmp\kc\discbullet{stconjectured}\fi
    \def\ks{s}\ifx\tmp\ks\discbullet{ststated}\fi
    \def\ko{o}\ifx\tmp\ko\discbullet{stoutside}\fi
    \endgroup
  \fi
}

% --- auto-count totals per v/p/h/c key (script's s/o are not tallied).
%     totcount persists each grand total to the .aux, so headers rendered
%     before the blocks print correct counts after latexmk's multi-pass run.
\usepackage{totcount}
\newtotcounter{proto@v}\newtotcounter{proto@p}%
\newtotcounter{proto@h}\newtotcounter{proto@c}
\newcommand{\protototal}[1]{%
  \ifcsname c@#1@totc\endcsname \totvalue{#1}\else 0\fi}
\newcommand{\protoTotV}{\protototal{proto@v}}
\newcommand{\protoTotP}{\protototal{proto@p}}
\newcommand{\protoTotH}{\protototal{proto@h}}
\newcommand{\protoTotC}{\protototal{proto@c}}
\newcommand{\protocount}[1]{%
  \def\tmp{#1}%
  \def\kv{v}\ifx\tmp\kv\stepcounter{proto@v}\fi
  \def\kp{p}\ifx\tmp\kp\stepcounter{proto@p}\fi
  \def\kh{h}\ifx\tmp\kh\stepcounter{proto@h}\fi
  \def\kc{c}\ifx\tmp\kc\stepcounter{proto@c}\fi}

% Head wrapper: bump the count, then disc + thin space, only when non-empty.
\newcommand{\dischead}[1]{\ifx\\#1\\\else\protocount{#1}\disc{#1}\thinspace\fi}

% Custom style: the note (#3 = the disc key) renders FIRST as a disc.
\newtheoremstyle{status}%
  {\topsep}{\topsep}%
  {\itshape}%
  {}%
  {\bfseries}%
  {.}%
  {.5em}%
  {\dischead{#3}\thmname{#1}\thmnumber{ #2}}%
\theoremstyle{status}

\newtheorem{remark}{Remark}

\newenvironment{myproof}[2]{\paragraph{Proof of {#1} {#2}.}}{\hfill $\blacksquare$}

% \blocktag{<text>}: thin rule + small bold tag, so each block reads as a card.
\newcommand{\blocktag}[1]{%
  \par\bigskip
  {\color{gray!55}\rule{\linewidth}{0.4pt}}\par
  \nobreak\vspace{2pt}%
  {\footnotesize\bfseries\color{gray!70}#1}\par
  \nobreak\vspace{2pt}%
}

\newcommand{\Proofsketch}{\noindent\textbf{Proof sketch.}\ }
\newcommand{\Risk}{\noindent\textbf{Risk.}\ }
\newcommand{\Uses}{\noindent\textbf{Uses $\rightarrow$}\ }
\newcommand{\Next}{\noindent\textbf{Next $\rightarrow$}\ }

\newcommand{\Role}{\noindent\textbf{Role.}\ }

% Compact headers (no cover page — the body starts on page 1)
% =====================================================================
\newcounter{proto@scratch}

% \stagehead — for stage/restage. Reads: \protonum \protofield \protodate
% \prototitle \protosubtitle \protopunch \protospine, plus the auto-tally.
\newcommand{\stagehead}{%
\begingroup
\setlength{\parindent}{0pt}%
{\footnotesize\color{gray!75}%
  \textbf{\textnumero\,\protonum}\;\textperiodcentered\;\protofield
  \;\textperiodcentered\;\protodate
  \hfill \textsc{paper-theater \textperiodcentered\ stage \textperiodcentered\ v1.1}}\par
\vspace{4pt}{\color{gray!40}\rule{\linewidth}{0.4pt}}\par\vspace{8pt}
{\LARGE\bfseries \prototitle\par}\vspace{2pt}
{\itshape\color{gray!75}\protosubtitle\par}\vspace{8pt}
{\color{stplausible}$\blacktriangleright$}\ \protopunch\par\vspace{6pt}
{\footnotesize\textbf{SPINE.}\ \protospine\par}\vspace{6pt}
{\footnotesize
  \setcounter{proto@scratch}{\protoTotV}\disc{v}\,\arabic{proto@scratch}~Verified
  \;\textperiodcentered\;
  \setcounter{proto@scratch}{\protoTotP}\disc{p}\,\arabic{proto@scratch}~Plausible
  \;\textperiodcentered\;
  \setcounter{proto@scratch}{\protoTotH}\disc{h}\,\arabic{proto@scratch}~Heuristic
  \;\textperiodcentered\;
  \setcounter{proto@scratch}{\protoTotC}\disc{c}\,\arabic{proto@scratch}~Conjectured
  \quad{\color{gray!70}--- run \textsc{tryout} to charge discs green}\par}
\vspace{4pt}{\color{gray!40}\rule{\linewidth}{0.4pt}}\par
\endgroup\bigskip}

% \scripthead — for script on EXTERNAL papers. Reads: \prototitle (paper
% title), \protosubtitle (authors \textperiodcentered\ venue
% \textperiodcentered\ year), \protodate. s/o markers only.
\newcommand{\scripthead}{%
\begingroup
\setlength{\parindent}{0pt}%
{\footnotesize\color{gray!75}%
  scripted \protodate
  \hfill \textsc{paper-theater \textperiodcentered\ script \textperiodcentered\ v1.1}}\par
\vspace{4pt}{\color{gray!40}\rule{\linewidth}{0.4pt}}\par\vspace{8pt}
{\LARGE\bfseries \prototitle\par}\vspace{2pt}
{\itshape\color{gray!75}\protosubtitle\par}\vspace{6pt}
{\footnotesize \disc{s}~\textbf{Stated} --- proved in the paper (not re-verified here)
 \;\textperiodcentered\;
 \disc{o}~\textbf{Outside} --- cited/imported from elsewhere\par}
\vspace{4pt}{\color{gray!40}\rule{\linewidth}{0.4pt}}\par
\endgroup\bigskip}

% \scriptheadown — for script on the AUTHOR'S OWN draft: v/p/h/c discs are
% OUR verification state (a claimed proof enters at p, never v), with the
% auto-tally strip. Added in v1.1 after the first own-draft run had to
% hand-roll this header.
\newcommand{\scriptheadown}{%
\begingroup
\setlength{\parindent}{0pt}%
{\footnotesize\color{gray!75}%
  scripted \protodate
  \hfill \textsc{paper-theater \textperiodcentered\ script \textperiodcentered\ v1.1}}\par
\vspace{4pt}{\color{gray!40}\rule{\linewidth}{0.4pt}}\par\vspace{8pt}
{\LARGE\bfseries \prototitle\par}\vspace{2pt}
{\itshape\color{gray!75}\protosubtitle\par}\vspace{6pt}
{\footnotesize\textbf{Author's own draft} --- discs are \emph{our} verification
 state, not the paper's claims.\;
 \disc{v}~Verified \textperiodcentered\ \disc{p}~Plausible (claimed-proved, not
 re-verified) \textperiodcentered\ \disc{h}~Heuristic \textperiodcentered\
 \disc{c}~Conjectured.\\[2pt]
 \setcounter{proto@scratch}{\protoTotV}\disc{v}\,\arabic{proto@scratch}~Verified
 \;\textperiodcentered\;
 \setcounter{proto@scratch}{\protoTotP}\disc{p}\,\arabic{proto@scratch}~Plausible
 \;\textperiodcentered\;
 \setcounter{proto@scratch}{\protoTotH}\disc{h}\,\arabic{proto@scratch}~Heuristic
 \;\textperiodcentered\;
 \setcounter{proto@scratch}{\protoTotC}\disc{c}\,\arabic{proto@scratch}~Conjectured\par}
\vspace{4pt}{\color{gray!40}\rule{\linewidth}{0.4pt}}\par
\endgroup\bigskip}
